.class %map_enumerator
	.field map
	.field pos
	.field len
	.method تقدم 1 1
		popl this
		
		pushl this
		pushl this
		getfld pos
		pushv 1
		add
		setfld pos
		
		pushl this
		getfld pos
		pushl this
		getfld len
		gt
		if
		jmp domove
		pushv false
		ret
		domove:
		pushv true
		ret
	.endmethod
	.method القيمة.الحالية 1 1
		popl this
		pushl this
		getfld pos
		pushl this
		getfld map
		callex mapkey
		popl %key
		new %pair
		popl ret
		pushl ret
		pushl %key
		setfld مفتاح
		pushl ret
		pushl this
		getfld map
		pushl %key
		getarr
		setfld قيمة
		pushl ret
		ret
	.endmethod
.endclass
